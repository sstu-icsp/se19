package po.pkg0;

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class PO0 {

    public static void main(String[] args) {
        ArrayList<String> sym = new ArrayList<String>() {};
        for(char c = 'A'; c <= 'z'; c++)
        {
            String s = new String();
            s += c;
            sym.add(s);
        }
        for(int c = 0; c < 10; c++)
        {
            String s = new String();
            s += c;
            sym.add(s);
        }
        sym.add("_");
        boolean flag = true;
        Scanner scan = new Scanner(System.in, "Cp1251");
        while(true)
        {       
            System.out.println("Введите идентификатор на языке C:");
            String identifier = new String();
            identifier = scan.nextLine();
            if(Character.isDigit(identifier.charAt(0))) 
            {
                System.out.println("Нет");
            }
            else 
            {
                char[] ch = identifier.toCharArray();
                for (char el : ch) {
                    if(sym.contains(Character.toString(el)));
                    else 
                    {
                        flag = false;
                        break;
                    }
                }
                if(flag) System.out.println("Да");
                else System.out.println("Нет");
            }
        }
    }
    
}
